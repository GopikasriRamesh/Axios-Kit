Tabs

Tabs allow users to switch between different sections of content within the same context. This pure CSS version uses radio inputs for state management.

<div class="tabs">
  <!-- Tab 1 -->
  <input type="radio" name="my-tabs" id="tab1" class="tabs__input" checked>
  <label for="tab1" class="tabs__label">Tab One</label>
  <div class="tabs__panel">
    <p>Content for the first tab.</p>
  </div>

  <!-- Tab 2 -->
  <input type="radio" name="my-tabs" id="tab2" class="tabs__input">
  <label for="tab2" class="tabs__label">Tab Two</label>
  <div class="tabs__panel">
    <p>Content for the second tab.</p>
  </div>
</div>
.tabs__input {
  display: none; /* Hide the actual radio button */
}
.tabs__label {
  padding: var(--space-sm);
  cursor: pointer;
  border-bottom: 2px solid var(--border-color);
}
.tabs__panel {
  display: none; /* Hide all panels by default */
  padding: var(--space-md);
  border-top: 2px solid var(--border-color);
}
/* Show the panel that is a sibling of the checked input */
.tabs__input:checked + .tabs__label + .tabs__panel {
  display: block;
}
/* Style the active tab label */
.tabs__input:checked + .tabs__label {
  border-bottom-color: var(--primary-text);
  font-weight: 700;
}
// While the pure CSS version works well,
// a JavaScript approach is often more
// flexible for complex applications.

const tabs = document.querySelectorAll('.tab-button');
const panels = document.querySelectorAll('.tab-panel');

tabs.forEach(tab => {
  tab.addEventListener('click', () => {
    const target = tab.dataset.target; // e.g., "#panel1"

    // Deactivate all tabs and panels
    tabs.forEach(t => t.classList.remove('active'));
    panels.forEach(p => p.classList.remove('active'));

    // Activate the clicked tab and its panel
    tab.classList.add('active');
    document.querySelector(target).classList.add('active');
  });
});
