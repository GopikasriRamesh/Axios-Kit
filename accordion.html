Accordion

Accordions allow users to toggle the visibility of content sections. This pure CSS version uses hidden checkboxes for state management and `max-height` for a smooth animation.

<div class="accordion">
  <!-- Accordion Item -->
  <div class="accordion__item">
    <input type="checkbox" name="my-accordion" id="accordion1" class="accordion__input">
    <label for="accordion1" class="accordion__label">Section One</label>
    <div class="accordion__content">
      <p>Content for the first section.</p>
    </div>
  </div>

  <!-- Another Item (open by default) -->
  <div class="accordion__item">
    <input type="checkbox" name="my-accordion" id="accordion2" class="accordion__input" checked>
    <label for="accordion2" class="accordion__label">Section Two</label>
    <div class="accordion__content">
      <p>Content for the second section.</p>
    </div>
  </div>
</div>
.accordion__input {
  display: none; /* Hide the checkbox */
}
.accordion__label {
  display: block;
  padding: var(--space-sm);
  cursor: pointer;
  /* ... more styles in main.css ... */
}
.accordion__content {
  overflow: hidden;
  max-height: 0; /* Hidden by default */
  transition: max-height 0.3s ease-out;
}
/* Show the content when the checkbox is checked */
.accordion__input:checked ~ .accordion__content {
  max-height: 1000px; /* Animate to a large height */
}
// For a more accessible accordion, a
// JavaScript approach is recommended.

const accordionLabels = document.querySelectorAll('.accordion__label');

accordionLabels.forEach(label => {
  label.addEventListener('click', () => {
    // Toggle the 'active' class on the parent item
    const item = label.parentElement;
    item.classList.toggle('active');

    // Toggle the content's visibility
    const content = label.nextElementSibling;
    if (content.style.maxHeight) {
      content.style.maxHeight = null;
    } else {
      content.style.maxHeight = content.scrollHeight + "px";
    } 
  });
});
